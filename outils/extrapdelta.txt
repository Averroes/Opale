// fonction qui extrapole une série x par la variation de y entre une date 1 
// et une date 2
// en entrée : serie x, serie y 
//             date1 (i.e. une période avant la première nouvelle valeur) date2

addfun main;

PROCEDURE main(x,y,debut,fin)
  {
  	serie=SUBRANGE (x,STARTDATE(x), debut);
 t1=debut;
 t2=fin;
 
  for (i=t1;i<=t2-1;i=i+1)
     {
     valeur=serie[i]+y[i+1]-y[i];
     serie=( c.(serie,valeur) );
     
     }
  return(serie);
  }
