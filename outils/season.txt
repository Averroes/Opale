// crée des dummies saisonnières
// en entrée le trimestre choisi (1,2,3 ou 4)
// la date de début et celle de fin


addfun main;
procedure main(t,debut,fin)

{
a=t;

u=0;
if (date2frac(debut)==a) then u=1 ;
b=reshape(u,debut);
for (i=debut+1;i<=fin;i=i+1)
	begin;
	u=0;
	if (date2frac(i)==a) then u=1 ;
	b=c.(b,u);
	end;

return (b);

}
