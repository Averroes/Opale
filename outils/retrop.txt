// fonction qui rétropole une série x par le taux de croissance de y entre la date début de y 
// et la date de début de x
// en entrée : serie x, serie y définie sur une période allant plus loin dans le passé

addfun main;

procedure main(x,y)
  {
  t1=startdate(x);
  t2=startdate(y);
 serie=x;
  for (i=t1-1;i>=t2;i=i-1)
     {j=i+1;
     valeur=reshape(serie[j]*y[i]/y[j],i);
     serie=( c.(valeur,serie) );
     
     }
  return(serie);
  }
